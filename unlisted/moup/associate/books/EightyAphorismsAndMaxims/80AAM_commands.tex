\newdimen\secheight
\secheight = 2em

\newcommand{\secdec}[2]{
	\raisebox{-0.25\secheight}{
		\begin{tikzpicture}
		\node[anchor = north, rotate=#1] at (0,0) {\pgfornament[opacity=0.50, height=\secheight]{#2}};
		\end{tikzpicture}
	}
}



\newcommand{\aphorism}[2]{
	\eachpageornament
	\section*{
		\hspace*{\fill}
		\secdec{90}{26}
		\hspace*{0.5ex}
		#1
		\hspace*{0.5ex}
		\secdec{-90}{26}
		\hspace*{\fill}
	}
	\vspace*{\fill}
	#2
	\vspace*{\fill}
	\pagebreak
}

\newcommand{\aphorismtwo}[3]{
	\eachpageornament
	\section*{
		\hspace*{\fill}
		\secdec{90}{26}
		\hspace*{0.5ex}
		#1
		\hspace*{0.5ex}
		\secdec{-90}{26}
		\hspace*{\fill}
	}
	\vspace*{\fill}
	{\textbf{\LARGE#2}}{#3}
	\vspace*{\fill}
	\pagebreak
}
